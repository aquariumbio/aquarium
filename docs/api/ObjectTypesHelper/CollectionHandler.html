def matrix item

  begin
    m = JSON.parse item.data, symbolize_names: true
  rescue Exception => e
    m = nil
  end

  if m.class == Hash && m[:matrix] && m[:matrix].class == Array && m[:matrix].length > 0 && m[:matrix][0].class == Array
    m[:matrix]
  else
    [[]]
  end

end

def new_item params

  r = params[:item][:rows].to_i
  c = params[:item][:cols].to_i

  item = Item.make({
      quantity: 1,
      inuse: 0,
      data: { matrix: Array.new(r,Array.new(c,-1)) }.to_json
    },object_type: @object_type)

  item.location = params[:item][:location]
  item

end
